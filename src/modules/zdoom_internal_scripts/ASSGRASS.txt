str grassFlatz[4] =
{
	"GRASS1",
	"GRASS2",
	"RROCK19",
	"RROCK20",
};

str snowFlatz[11] =
{
	"SNOW1",
	"SNOW5",
	"SNOW6",
	"SNOW7",
	"SNOW8",
	"SNOW10F",
	"SNOW11F",
	"SNOW12F",
	"SNOW13F",
	"SNOW14F",
	"SNOW15F",
};

str sandFlatz[7] =
{
	"SAND1",
	"SAND2",
	"SAND3",
	"SAND4",
	"SAND5",
	"SAND6",
	"SAND7",
};

str hellFlatz[14] =
{
	"RROCK01",
	"RROCK02",
	"RROCK05",
	"RROCK06",
	"RROCK07",
	"RROCK08",
	"FLOOR6_1",
	"SFLR6_1",
	"SFLR6_4",
	"SKINFLT1",
	"SLIME09",
	"SLIME10",
	"SLIME11",
	"SLIME12",
};

Script "IsMyAssGrass" (void)
{


	bool onGrass = false;
	bool onSnow = false;
	bool onSand = false;
	bool onHellscape = false;
	int flatIndex;

	for ( flatIndex = 0; flatIndex < 4; flatIndex++ )
	{
		if (CheckActorFloorTexture(0, grassFlatz[flatIndex]))
		{
			onGrass = true;
		}
		Delay (1);
	}

	for ( flatIndex = 0; flatIndex < 11; flatIndex++ )
	{
		if (CheckActorFloorTexture(0, snowFlatz[flatIndex]))
		{
			onSnow = true;
		}
		Delay (1);
	}

	for ( flatIndex = 0; flatIndex < 7; flatIndex++ )
	{
		if (CheckActorFloorTexture(0, sandFlatz[flatIndex]))
		{
			onSand = true;
		}
		Delay (1);
	}

	for ( flatIndex = 0; flatIndex < 14; flatIndex++)
	{
		if (CheckActorFloorTexture(0, hellFlatz[flatIndex]))
		{
			onHellscape = true;
		}
		Delay (1);
	}

	if (onGrass)
	{
		GiveInventory("AssIsGrass", 1);
	}
	else if (onSnow)
	{
		GiveInventory("AssIsSnow", 1);
	}
	else if (onSand)
	{
		GiveInventory("AssIsSand", 1);
	}
	else if (onHellscape)
	{
		GiveInventory("AssIsHell", 1);
	}
	else
	{
		GiveInventory("AssIsUnknown", 1);
		terminate;
	}
}
