#library "ASSGRASS"

#include "zcommon.acs"

Script "IsMyAssGrass" (void)
{
	str flatz[4] =
	{
		"GRASS1",
		"GRASS2",
		"RROCK19",
		"RROCK20",
	};

	bool onGrass = false;

	for ( int flat_index = 1; flat_index <= 4; flat_index++ )
	{
		if (CheckActorFloorTexture(0, flatz[flat_index]))
		{
			onGrass = true;
		}
	}

	if (onGrass)
	{
		GiveInventory("AssIsGrass", 1);
	}
	else
	{
		GiveInventory("AssIsUnknown", 1);
		terminate;
	}
}
