actor FancyObligeTree replaces BigTree
{
	Radius 16
	Height 128
	ProjectilePassHeight -16
	+SOLID

	States
	{
		Spawn:
			TRE2 A 3
			TRE2 A 0 ACS_NamedExecuteAlways("IsMyAssGrass")
			Goto Decide

		Decide:
			TRE2 A 1
			TRE2 A 0 A_JumpIfInventory("AssIsGrass", 1, "OnGrass")
			TRE2 A 0 A_JumpIfInventory("AssIsUnknown", 1, "Vanilla")

		OnGrass:
			TRE2 A 0 A_Jump(255, "OakTree", "RedwoodTree", "SomeThinTree")

		OakTree:
			OAK1 A -1

		RedwoodTree:
			RED1 A -1

		SomeThinTree:
			THN1 A -1

		Vanilla:
			TRE2 A -1
	}
}
